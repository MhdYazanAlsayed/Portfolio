public class EmployeeService
{
    private readonly HRDbContext _context;
    private readonly ILogger _logger;

    public EmployeeService(HRDbContext context, ILogger logger)
    {
        _context = context;
        _logger = logger;
    }

    public async Task> CreateEmployeeAsync(CreateEmployeeDto dto)
    {
        try
        {
            // Validate employee data
            var validationResult = await ValidateEmployeeDataAsync(dto);
            if (!validationResult.IsValid)
            {
                return Result.Failure(validationResult.Errors);
            }

            // Check for duplicate employee ID
            var existingEmployee = await _context.Employees
                .FirstOrDefaultAsync(e => e.EmployeeId == dto.EmployeeId);
            
            if (existingEmployee != null)
            {
                return Result.Failure("Employee ID already exists");
            }

            // Create new employee
            var employee = new Employee
            {
                EmployeeId = dto.EmployeeId,
                FirstName = dto.FirstName,
                LastName = dto.LastName,
                Email = dto.Email,
                DepartmentId = dto.DepartmentId,
                Position = dto.Position,
                HireDate = dto.HireDate,
                Salary = dto.Salary,
                IsActive = true,
                CreatedAt = DateTime.UtcNow
            };

            _context.Employees.Add(employee);
            await _context.SaveChangesAsync();

            _logger.LogInformation("Employee created successfully: {EmployeeId}", employee.EmployeeId);
            return Result.Success(employee);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error creating employee");
            return Result.Failure("An error occurred while creating the employee");
        }
    }

    public async Task> UpdateEmployeeAsync(int id, UpdateEmployeeDto dto)
    {
        try
        {
            var employee = await _context.Employees.FindAsync(id);
            if (employee == null)
            {
                return Result.Failure("Employee not found");
            }

            // Update employee properties
            employee.FirstName = dto.FirstName;
            employee.LastName = dto.LastName;
            employee.Email = dto.Email;
            employee.DepartmentId = dto.DepartmentId;
            employee.Position = dto.Position;
            employee.Salary = dto.Salary;
            employee.UpdatedAt = DateTime.UtcNow;

            await _context.SaveChangesAsync();

            _logger.LogInformation("Employee updated successfully: {EmployeeId}", employee.EmployeeId);
            return Result.Success(employee);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error updating employee");
            return Result.Failure("An error occurred while updating the employee");
        }
    }

    public async Task> DeleteEmployeeAsync(int id)
    {
        try
        {
            var employee = await _context.Employees.FindAsync(id);
            if (employee == null)
            {
                return Result.Failure("Employee not found");
            }

            // Soft delete - mark as inactive
            employee.IsActive = false;
            employee.DeletedAt = DateTime.UtcNow;

            await _context.SaveChangesAsync();

            _logger.LogInformation("Employee deleted successfully: {EmployeeId}", employee.EmployeeId);
            return Result.Success(true);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error deleting employee");
            return Result.Failure("An error occurred while deleting the employee");
        }
    }

    public async Task>> GetEmployeesAsync(EmployeeFilterDto filter)
    {
        try
        {
            var query = _context.Employees
                .Include(e => e.Department)
                .Where(e => e.IsActive);

            // Apply filters
            if (!string.IsNullOrEmpty(filter.SearchTerm))
            {
                query = query.Where(e => 
                    e.FirstName.Contains(filter.SearchTerm) ||
                    e.LastName.Contains(filter.SearchTerm) ||
                    e.EmployeeId.Contains(filter.SearchTerm));
            }

            if (filter.DepartmentId.HasValue)
            {
                query = query.Where(e => e.DepartmentId == filter.DepartmentId.Value);
            }

            if (filter.IsActive.HasValue)
            {
                query = query.Where(e => e.IsActive == filter.IsActive.Value);
            }

            // Apply sorting
            query = filter.SortBy?.ToLower() switch
            {
                "name" => filter.SortDescending ? query.OrderByDescending(e => e.FirstName) : query.OrderBy(e => e.FirstName),
                "department" => filter.SortDescending ? query.OrderByDescending(e => e.Department.Name) : query.OrderBy(e => e.Department.Name),
                "hiredate" => filter.SortDescending ? query.OrderByDescending(e => e.HireDate) : query.OrderBy(e => e.HireDate),
                _ => query.OrderBy(e => e.FirstName)
            };

            var totalCount = await query.CountAsync();
            var employees = await query
                .Skip((filter.Page - 1) * filter.PageSize)
                .Take(filter.PageSize)
                .ToListAsync();

            var pagedResult = new PagedResult
            {
                Items = employees,
                TotalCount = totalCount,
                Page = filter.Page,
                PageSize = filter.PageSize,
                TotalPages = (int)Math.Ceiling((double)totalCount / filter.PageSize)
            };

            return Result>.Success(pagedResult);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error retrieving employees");
            return Result>.Failure("An error occurred while retrieving employees");
        }
    }

    private async Task ValidateEmployeeDataAsync(CreateEmployeeDto dto)
    {
        var result = new ValidationResult();

        if (string.IsNullOrWhiteSpace(dto.EmployeeId))
            result.Errors.Add("Employee ID is required");

        if (string.IsNullOrWhiteSpace(dto.FirstName))
            result.Errors.Add("First name is required");

        if (string.IsNullOrWhiteSpace(dto.LastName))
            result.Errors.Add("Last name is required");

        if (string.IsNullOrWhiteSpace(dto.Email) || !IsValidEmail(dto.Email))
            result.Errors.Add("Valid email is required");

        if (dto.DepartmentId <= 0)
            result.Errors.Add("Valid department is required");

        if (dto.Salary < 0)
            result.Errors.Add("Salary cannot be negative");

        result.IsValid = !result.Errors.Any();
        return result;
    }

    private bool IsValidEmail(string email)
    {
        try
        {
            var addr = new System.Net.Mail.MailAddress(email);
            return addr.Address == email;
        }
        catch
        {
            return false;
        }
    }
}
OrderService.cs

Service for managing restaurant orders with inventory tracking and real-time updates

C# 


public class OrderService
{
    private readonly RestaurantDbContext _context;
    private readonly IInventoryService _inventoryService;
    private readonly ILogger _logger;

    public OrderService(RestaurantDbContext context, IInventoryService inventoryService, ILogger logger)
    {
        _context = context;
        _inventoryService = inventoryService;
        _logger = logger;
    }

    public async Task> CreateOrderAsync(CreateOrderDto dto)
    {
        using var transaction = await _context.Database.BeginTransactionAsync();
        
        try
        {
            // Validate order items and check inventory
            var inventoryValidation = await ValidateInventoryAsync(dto.OrderItems);
            if (!inventoryValidation.IsValid)
            {
                return Result.Failure(inventoryValidation.Errors);
            }

            // Create order
            var order = new Order
            {
                OrderNumber = await GenerateOrderNumberAsync(),
                CustomerId = dto.CustomerId,
                TableNumber = dto.TableNumber,
                OrderDate = DateTime.UtcNow,
                Status = OrderStatus.Pending,
                TotalAmount = 0,
                CreatedBy = dto.CreatedBy
            };

            _context.Orders.Add(order);
            await _context.SaveChangesAsync();

            // Add order items
            decimal totalAmount = 0;
            foreach (var itemDto in dto.OrderItems)
            {
                var menuItem = await _context.MenuItems.FindAsync(itemDto.MenuItemId);
                if (menuItem == null)
                {
                    await transaction.RollbackAsync();
                    return Result.Failure($"Menu item not found: {itemDto.MenuItemId}");
                }

                var orderItem = new OrderItem
                {
                    OrderId = order.Id,
                    MenuItemId = itemDto.MenuItemId,
                    Quantity = itemDto.Quantity,
                    UnitPrice = menuItem.Price,
                    TotalPrice = menuItem.Price * itemDto.Quantity,
                    SpecialInstructions = itemDto.SpecialInstructions
                };

                _context.OrderItems.Add(orderItem);
                totalAmount += orderItem.TotalPrice;

                // Update inventory
                await _inventoryService.ReduceInventoryAsync(itemDto.MenuItemId, itemDto.Quantity);
            }

            // Update order total
            order.TotalAmount = totalAmount;
            _context.Orders.Update(order);

            await _context.SaveChangesAsync();
            await transaction.CommitAsync();

            _logger.LogInformation("Order created successfully: {OrderNumber}", order.OrderNumber);
            return Result.Success(order);
        }
        catch (Exception ex)
        {
            await transaction.RollbackAsync();
            _logger.LogError(ex, "Error creating order");
            return Result.Failure("An error occurred while creating the order");
        }
    }

    public async Task> UpdateOrderStatusAsync(int orderId, OrderStatus newStatus)
    {
        try
        {
            var order = await _context.Orders
                .Include(o => o.OrderItems)
                .FirstOrDefaultAsync(o => o.Id == orderId);

            if (order == null)
            {
                return Result.Failure("Order not found");
            }

            var oldStatus = order.Status;
            order.Status = newStatus;
            order.UpdatedAt = DateTime.UtcNow;

            // If order is cancelled, restore inventory
            if (newStatus == OrderStatus.Cancelled && oldStatus != OrderStatus.Cancelled)
            {
                await RestoreInventoryAsync(order.OrderItems);
            }

            await _context.SaveChangesAsync();

            _logger.LogInformation("Order status updated: {OrderNumber} from {OldStatus} to {NewStatus}", 
                order.OrderNumber, oldStatus, newStatus);

            return Result.Success(order);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error updating order status");
            return Result.Failure("An error occurred while updating the order status");
        }
    }

    public async Task> AddOrderItemAsync(int orderId, AddOrderItemDto dto)
    {
        using var transaction = await _context.Database.BeginTransactionAsync();
        
        try
        {
            var order = await _context.Orders
                .Include(o => o.OrderItems)
                .FirstOrDefaultAsync(o => o.Id == orderId);

            if (order == null)
            {
                return Result.Failure("Order not found");
            }

            if (order.Status != OrderStatus.Pending)
            {
                return Result.Failure("Cannot add items to an order that is not pending");
            }

            // Check inventory
            var inventoryCheck = await _inventoryService.CheckAvailabilityAsync(dto.MenuItemId, dto.Quantity);
            if (!inventoryCheck.IsAvailable)
            {
                return Result.Failure($"Insufficient inventory: {inventoryCheck.Message}");
            }

            var menuItem = await _context.MenuItems.FindAsync(dto.MenuItemId);
            var orderItem = new OrderItem
            {
                OrderId = orderId,
                MenuItemId = dto.MenuItemId,
                Quantity = dto.Quantity,
                UnitPrice = menuItem.Price,
                TotalPrice = menuItem.Price * dto.Quantity,
                SpecialInstructions = dto.SpecialInstructions
            };

            _context.OrderItems.Add(orderItem);
            
            // Update order total
            order.TotalAmount += orderItem.TotalPrice;
            order.UpdatedAt = DateTime.UtcNow;

            // Update inventory
            await _inventoryService.ReduceInventoryAsync(dto.MenuItemId, dto.Quantity);

            await _context.SaveChangesAsync();
            await transaction.CommitAsync();

            _logger.LogInformation("Order item added successfully to order: {OrderNumber}", order.OrderNumber);
            return Result.Success(order);
        }
        catch (Exception ex)
        {
            await transaction.RollbackAsync();
            _logger.LogError(ex, "Error adding order item");
            return Result.Failure("An error occurred while adding the order item");
        }
    }

    private async Task ValidateInventoryAsync(List orderItems)
    {
        var result = new ValidationResult();

        foreach (var item in orderItems)
        {
            var availability = await _inventoryService.CheckAvailabilityAsync(item.MenuItemId, item.Quantity);
            if (!availability.IsAvailable)
            {
                result.Errors.Add($"Insufficient inventory for menu item {item.MenuItemId}: {availability.Message}");
            }
        }

        result.IsValid = !result.Errors.Any();
        return result;
    }

    private async Task GenerateOrderNumberAsync()
    {
        var today = DateTime.UtcNow.Date;
        var orderCount = await _context.Orders
            .Where(o => o.OrderDate.Date == today)
            .CountAsync();

        return $"ORD-{today:yyyyMMdd}-{(orderCount + 1):D4}";
    }

    private async Task RestoreInventoryAsync(List orderItems)
    {
        foreach (var item in orderItems)
        {
            await _inventoryService.IncreaseInventoryAsync(item.MenuItemId, item.Quantity);
        }
    }
}
